    public List<Employee> fetchEmployeesFromApi() throws ApiException {

        // 1. Zbuduj zapytanie (Request) - dokĹ‚adnie jak na wykĹ‚adzie
        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create("https://jsonplaceholder.typicode.com/users"))
                .GET() // MĂłwimy, ĹĽe to metoda GET
                .build();

                // JeĹ›li serwer nie odpowiedziaĹ‚ "OK" (kod 200), to jest bĹ‚Ä…d

            JsonArray usersArray = gson.fromJson(jsonBody, JsonArray.class);

                // najpierw pobraÄ‡ obiekt "company", a dopiero z niego "name"

                        .get("name")
                        .getAsString();

                Position position = Position.PROGRAMISTA;
                double salary = position.getBaseSalary(); // Bierzemy bazowÄ… stawkÄ™

                // Tworzymy pracownika i dodajemy do listy

            throw new ApiException("BĹ‚Ä…d parsowania odpowiedzi JSON: " + e.getMessage(), e);
